function Floyd_Warshall ()

for each edge (u,v)

dist[u][v] <- w(u,v)

next[u][v] <- v

for k from 1 to |V|

for i from 1 to |V|

for j from 1 to |V|

if dist[i][j] > dist[i][k] + dist[k][j] then

dist[i][j] <- dist[i][k] + dist[k][j]

next[i][j] <- next[i][k]
